# CMakeLists.txt to build is900cf

INCLUDE_DIRECTORIES (${DTK_SOURCE_DIR}/include)
ADD_LIBRARY (is900cf MODULE is900cf.cpp)
TARGET_LINK_LIBRARIES (is900cf dtk)
SET_TARGET_PROPERTIES (is900cf PROPERTIES PREFIX "")
INSTALL (TARGETS is900cf LIBRARY DESTINATION etc/dtk/serviceDSO PERMISSIONS
	OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ
	WORLD_EXECUTE)
INSTALL (FILES VTCaveIS900.config VTWallIS900.config VTCaveIS900.txt VTWallIS900.txt  DESTINATION etc/dtk/serviceDSO
	PERMISSIONS OWNER_READ OWNER_WRITE GROUP_READ WORLD_READ)
