# CMakeLists.txt to build udpEcho

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include )
ADD_LIBRARY( udpEcho MODULE udpEcho.cpp )
TARGET_LINK_LIBRARIES( udpEcho dtk )
ADD_EXECUTABLE( dtk-udpEchoWriter dtk-udpEchoWriter.cpp )
TARGET_LINK_LIBRARIES( dtk-udpEchoWriter dtk )
SET_TARGET_PROPERTIES( udpEcho PROPERTIES PREFIX "" )
INSTALL( TARGETS udpEcho dtk-udpEchoWriter RUNTIME DESTINATION bin LIBRARY
		DESTINATION etc/dtk/serviceDSO PERMISSIONS OWNER_READ OWNER_WRITE
		OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ WORLD_EXECUTE )
