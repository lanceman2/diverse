# CMakeLists.txt to build augmentDSO/dtkGLX

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include ${GLX_HEADER_PATH}
		${DTK_X11_INC_DIRS} )
LINK_DIRECTORIES( ${DTK_X11_LINK_DIRS} )

ADD_LIBRARY( gears MODULE gears.cpp )
TARGET_LINK_LIBRARIES( gears dtkGLX )

ADD_LIBRARY( glxSimpleDisplay MODULE glxSimpleDisplay.cpp )
TARGET_LINK_LIBRARIES( glxSimpleDisplay dtkGLX )

SET_TARGET_PROPERTIES( gears glxSimpleDisplay PROPERTIES PREFIX "" )
INSTALL( TARGETS gears glxSimpleDisplay LIBRARY DESTINATION etc/dtk/augmentDSO
		PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE
		WORLD_READ WORLD_EXECUTE )
