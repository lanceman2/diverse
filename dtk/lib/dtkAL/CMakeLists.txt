# CMakeLists.txt to build dtkAL

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include/dtk ${DTK_SOURCE_DIR}/config
	${DTK_SOURCE_DIR}/include/dtkAL ${DTK_AL_INC_DIRS} )
LINK_DIRECTORIES( ${DTK_AL_LINK_DIRS} )
ADD_LIBRARY( dtkAL soundBuffer.cpp soundManager.cpp soundSource.cpp
	soundSyncedSources.cpp )
TARGET_LINK_LIBRARIES( dtkAL dtk ${DTK_AL_LIBS} )
SET_TARGET_PROPERTIES( dtkAL PROPERTIES VERSION ${LIBDTKAL_VERSION}
	SOVERSION ${LIBDTKAL_MAJOR_VERSION} LINK_FLAGS "${DTK_AL_LINK_FLAGS}" )
INSTALL( TARGETS dtkAL RUNTIME DESTINATION bin LIBRARY DESTINATION lib ARCHIVE DESTINATION lib
	PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE
	WORLD_READ WORLD_EXECUTE )
