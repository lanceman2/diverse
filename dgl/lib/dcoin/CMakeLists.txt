# CMakeLists.txt to build lib/dcoin

IF( WIN32 )
	ADD_DEFINITIONS( /DCOIN_DLL /DDCOIN_EXPORTS )
ENDIF( WIN32 )

INCLUDE_DIRECTORIES( ${DGL_SOURCE_DIR}/include ${DTK_INCLUDE}
		${GLX_HEADER_PATH} ${COIN_INCLUDE_DIR} ${PRODUCER_INCLUDE_DIR}
		${OPENTHREADS_INCLUDE_DIR} )
ADD_LIBRARY( dcoin dcoin.cpp )
TARGET_LINK_LIBRARIES( dcoin dgl )
SET_TARGET_PROPERTIES( dcoin PROPERTIES LINK_FLAGS
		"${DGL_LIB_FLAGS} ${DGL_COIN_LDFLAGS}" VERSION ${LIBDGL_VERSION}
		SOVERSION ${LIBDGL_MAJOR_VERSION} )
INSTALL( TARGETS dcoin RUNTIME DESTINATION bin LIBRARY DESTINATION lib ARCHIVE
		DESTINATION lib PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE
		GROUP_READ GROUP_EXECUTE WORLD_READ WORLD_EXECUTE )
