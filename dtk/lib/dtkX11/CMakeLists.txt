# CMakeLists.txt to build libdtkX11

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include/dtk ${DTK_SOURCE_DIR}/config
		${DTK_SOURCE_DIR}/include/dtkX11 ${X11_INCLUDE_DIR} )
LINK_DIRECTORIES( ${DTK_X11_LINK_DIRS} )
ADD_LIBRARY( dtkX11 xevent.cpp xkeyboard.cpp xwindowList.cpp )
TARGET_LINK_LIBRARIES( dtkX11 dtk ${DTK_X11_LIBS} )
SET_TARGET_PROPERTIES( dtkX11 PROPERTIES VERSION ${LIBDTKX11_VERSION}
		SOVERSION ${LIBDTKX11_MAJOR_VERSION} )
INSTALL( TARGETS dtkX11 RUNTIME DESTINATION bin LIBRARY DESTINATION lib
		ARCHIVE DESTINATION lib PERMISSIONS OWNER_READ OWNER_WRITE
		OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ WORLD_EXECUTE )
