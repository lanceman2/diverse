# CMakeLists.txt to build libdtkGLX

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include/dtk ${DTK_SOURCE_DIR}/config
		${DTK_SOURCE_DIR}/include/dtkGLX ${DTK_SOURCE_DIR}/include/dtkX11
		${X11_INCLUDE_DIR} ${GLX_HEADER_PATH} ${DTK_GL_INC_DIRS} )
LINK_DIRECTORIES( ${DTK_X11_LINK_DIRS} ${DTK_GL_LINK_DIRS} )
ADD_LIBRARY( dtkGLX glxsimpleDisplay.cpp )
TARGET_LINK_LIBRARIES( dtkGLX dtkX11 ${DTK_X11_LIBS} ${OPENGL_gl_LIB_NAME} ${DARWIN_GL_FIX_FLAGS} )
SET_TARGET_PROPERTIES( dtkGLX PROPERTIES VERSION ${LIBDTKGLX_VERSION}
		SOVERSION ${LIBDTKGLX_MAJOR_VERSION} )
INSTALL( TARGETS dtkGLX RUNTIME DESTINATION bin LIBRARY DESTINATION lib ARCHIVE DESTINATION lib
		PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE
		WORLD_READ WORLD_EXECUTE )
