# CMakeLists.txt to build programs/dgl/dgl-navControl

IF( WIN32 )
	SET( WIN32_EXTRA_LIBS "comctl32.lib ws2_32.lib" )
ENDIF( WIN32 )

INCLUDE_DIRECTORIES( ${DGL_SOURCE_DIR}/include ${DTK_INCLUDE}
		${GLX_HEADER_PATH} ${PRODUCER_INCLUDE_DIR} ${OPENTHREADS_INCLUDE_DIR}
		${DGL_FLTK_INC_DIRS} )
LINK_DIRECTORIES( ${DGL_FLTK_LINK_DIRS} )
ADD_EXECUTABLE( dgl-navControl dgl-navControl.cpp )
TARGET_LINK_LIBRARIES( dgl-navControl dgl ${DGL_FLTK_LIBS} )
SET_TARGET_PROPERTIES( dgl-navControl PROPERTIES
		LINK_FLAGS "${DGL_LIB_FLAGS} ${DGL_FLTK_LINK_FLAGS} ${WIN32_EXTRA_LIBS}" )
INSTALL( TARGETS dgl-navControl RUNTIME DESTINATION bin PERMISSIONS OWNER_READ
		OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ
		WORLD_EXECUTE )
