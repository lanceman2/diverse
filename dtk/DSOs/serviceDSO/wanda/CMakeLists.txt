# CMakeLists.txt to build wanda

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include )
ADD_LIBRARY( wanda MODULE wanda.cpp )
SET_TARGET_PROPERTIES( wanda PROPERTIES PREFIX "" )
TARGET_LINK_LIBRARIES( wanda dtk )
INSTALL( TARGETS wanda LIBRARY DESTINATION etc/dtk/serviceDSO PERMISSIONS
		OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE
		WORLD_READ WORLD_EXECUTE )
INSTALL( FILES wanda.config DESTINATION etc/dtk/serviceDSO PERMISSIONS
		OWNER_READ OWNER_WRITE GROUP_READ WORLD_READ )
