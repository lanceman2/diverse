# CMakeLists.txt to build libdtkFLTK

IF( WIN32 )
	ADD_DEFINITIONS( /DFL_DLL /DLIBDTKFLTK_EXPORTS )
ENDIF( WIN32 )

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include/dtk ${DTK_SOURCE_DIR}/config
		${DTK_SOURCE_DIR}/include/dtkFLTK ${GLX_HEADER_PATH}
		${DTK_FLTK_GL_INC_DIRS} )
LINK_DIRECTORIES( ${DTK_FLTK_GL_LINK_DIRS} )
ADD_LIBRARY( dtkFLTK fltkSimpleDisplay.cpp )
TARGET_LINK_LIBRARIES( dtkFLTK dtk ${DTK_FLTK_GL_LIBS} )
SET_TARGET_PROPERTIES( dtkFLTK PROPERTIES
		LINK_FLAGS "${DTK_FLTK_GL_LINK_FLAGS}"
		VERSION "${LIBDTKFLTK_VERSION}"
		SOVERSION "${LIBDTKFLTK_MAJOR_VERSION}" )
INSTALL( TARGETS dtkFLTK RUNTIME DESTINATION bin LIBRARY DESTINATION lib
		ARCHIVE DESTINATION lib	PERMISSIONS OWNER_READ OWNER_WRITE
		OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ WORLD_EXECUTE )
