# CMakeLists.txt to build programs/dgl/dgl-readState

INCLUDE_DIRECTORIES( ${DGL_SOURCE_DIR}/include ${DTK_INCLUDE}
		${PRODUCER_INCLUDE_DIR} ${OPENTHREADS_INCLUDE_DIR}
		${CURSES_INCLUDE_PATH} )
ADD_EXECUTABLE( dgl-readState dgl-readState.cpp )
TARGET_LINK_LIBRARIES( dgl-readState dtk )
SET_TARGET_PROPERTIES( dgl-readState PROPERTIES LINK_FLAGS
		${DGL_CURSES_LDFLAGS} )
INSTALL( TARGETS dgl-readState RUNTIME DESTINATION bin PERMISSIONS OWNER_READ
		OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ
		WORLD_EXECUTE )
