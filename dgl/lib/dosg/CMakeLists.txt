# CMakeLists.txt to build lib/libdosg

IF( WIN32 )
	ADD_DEFINITIONS( /DDOSG_EXPORTS )
ENDIF( WIN32 )

INCLUDE_DIRECTORIES( ${DGL_SOURCE_DIR}/include ${DTK_INCLUDE}
		${GLX_HEADER_PATH} ${PRODUCER_INCLUDE_DIR}
		${OPENTHREADS_INCLUDE_DIR} ${OSG_INCLUDE_DIR} )
ADD_LIBRARY( dosg DOSG.cpp DOSGUtil.cpp dosgClipCullNode.cpp dosgShmCoordNode.cpp )
TARGET_LINK_LIBRARIES( dosg dgl )
SET_TARGET_PROPERTIES( dosg PROPERTIES LINK_FLAGS
		"${DGL_LIB_FLAGS} ${DGL_OSG_LDFLAGS}" VERSION ${LIBDGL_VERSION}
		SOVERSION ${LIBDGL_MAJOR_VERSION} )
INSTALL( TARGETS dosg RUNTIME DESTINATION bin LIBRARY DESTINATION lib ARCHIVE
		DESTINATION lib PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE
		GROUP_READ GROUP_EXECUTE WORLD_READ WORLD_EXECUTE )
