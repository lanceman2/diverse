# CMakeLists.txt to build lib/dgl

IF( WIN32 )
	ADD_DEFINITIONS( /DDGL_EXPORTS )
ENDIF( WIN32 )

INCLUDE_DIRECTORIES( ${DGL_SOURCE_DIR}/include ${DTK_INCLUDE}
		${PRODUCER_INCLUDE_DIR} ${OPENTHREADS_INCLUDE_DIR}
		${GLX_HEADER_PATH} )
ADD_LIBRARY( dgl DGL.cpp cDGL.cpp dglAugment.cpp DGLDisplay.cpp DGLWindow.cpp
		DGLScreen.cpp DGLOrthoHandler.cpp DGLScreenHandler.cpp
		DGLPerspectiveHandler.cpp DGLViewport.cpp DProducerInputCallback.cpp
		DGLUtil.cpp DGLKeyboard.cpp DGLMouse.cpp DGLMouseButton.cpp
		DGLMessage.cpp DGLPostOffice.cpp DGLSimulator.cpp DGLCameraThread.cpp
)
TARGET_LINK_LIBRARIES( dgl ${DTK_LIB} )
SET_TARGET_PROPERTIES( dgl PROPERTIES LINK_FLAGS ${DGL_LIB_FLAGS} VERSION
		${LIBDGL_VERSION} SOVERSION ${LIBDGL_MAJOR_VERSION} )
INSTALL( TARGETS dgl RUNTIME DESTINATION bin LIBRARY DESTINATION lib ARCHIVE
		DESTINATION lib PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE
		GROUP_READ GROUP_EXECUTE WORLD_READ WORLD_EXECUTE )
