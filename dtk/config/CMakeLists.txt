# CMakeLists.txt to build dtk-config

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include/dtk ${DTK_SOURCE_DIR}/lib/dtk )
ADD_EXECUTABLE( dtk-config dtk-config.cpp )
TARGET_LINK_LIBRARIES( dtk-config dtk )
INSTALL( TARGETS dtk-config RUNTIME DESTINATION bin PERMISSIONS OWNER_READ
		OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ
		WORLD_EXECUTE )
