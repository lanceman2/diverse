# CMakeLists.txt to build pcwand

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include )
ADD_LIBRARY( pcwand MODULE pcwand.cpp )
TARGET_LINK_LIBRARIES( pcwand dtk )
SET_TARGET_PROPERTIES( pcwand PROPERTIES PREFIX "" )
INSTALL( TARGETS pcwand LIBRARY DESTINATION etc/dtk/serviceDSO PERMISSIONS
		OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE
		WORLD_READ WORLD_EXECUTE )
