# CMakeLists.txt to build IWB

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include )
ADD_LIBRARY( iwbButtons MODULE iwbButtons.cpp )
TARGET_LINK_LIBRARIES( iwbButtons dtk )
ADD_LIBRARY( iwbFastrak MODULE iwbFastrak.cpp )
TARGET_LINK_LIBRARIES( iwbFastrak dtk )
SET_TARGET_PROPERTIES( iwbButtons iwbFastrak PROPERTIES PREFIX "" )
INSTALL( TARGETS iwbButtons iwbFastrak LIBRARY DESTINATION etc/dtk/serviceDSO
		PERMISSIONS OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ
		GROUP_EXECUTE WORLD_READ WORLD_EXECUTE )
