# CMakeLists.txt to build programs/dosg/dosg-fly

INCLUDE_DIRECTORIES( ${DGL_SOURCE_DIR}/include ${DTK_INCLUDE}
		${GLX_HEADER_PATH} ${OSG_INCLUDE_PATH} ${PRODUCER_INCLUDE_DIR}
		${OPENTHREADS_INCLUDE_DIR} )
ADD_EXECUTABLE( dosg-fly dosg-fly.cpp usage.cpp processArgs.cpp )
TARGET_LINK_LIBRARIES( dosg-fly dosg )
SET_TARGET_PROPERTIES( dosg-fly PROPERTIES LINK_FLAGS
		"${DGL_OSG_LDFLAGS} ${DGL_LIB_FLAGS}" )
INSTALL( TARGETS dosg-fly RUNTIME DESTINATION bin PERMISSIONS OWNER_READ
		OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ
		WORLD_EXECUTE )
