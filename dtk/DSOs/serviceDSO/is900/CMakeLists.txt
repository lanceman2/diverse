INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include )
ADD_LIBRARY( is900 MODULE is900.cpp )
TARGET_LINK_LIBRARIES( is900 dtk )
SET_TARGET_PROPERTIES( is900 PROPERTIES PREFIX "" )
INSTALL( TARGETS is900 LIBRARY DESTINATION etc/dtk/serviceDSO PERMISSIONS
		OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE
		WORLD_READ WORLD_EXECUTE )
