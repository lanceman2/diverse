# CMakeLists.txt to build programs/dosg/dosg-bounds

INCLUDE_DIRECTORIES( ${DGL_SOURCE_DIR}/include ${DTK_INCLUDE}
		${GLX_HEADER_PATH} ${OSG_INCLUDE_DIR} ${PRODUCER_INCLUDE_DIR} )
ADD_EXECUTABLE( dosg-bounds dosg-bounds.cpp )
TARGET_LINK_LIBRARIES( dosg-bounds dgl dosg )
SET_TARGET_PROPERTIES( dosg-bounds PROPERTIES LINK_FLAGS
		"${DGL_OSG_LDFLAGS} ${DGL_LIB_FLAGS}" )
INSTALL( TARGETS dosg-bounds RUNTIME DESTINATION bin PERMISSIONS OWNER_READ
		OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ
		WORLD_EXECUTE )
