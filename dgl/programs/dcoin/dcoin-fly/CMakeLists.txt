# CMakeLists.txt to build programs/dcoin/dcoin-fly

IF( WIN32 )
	ADD_DEFINITIONS( /DCOIN_DLL )
ENDIF( WIN32 )

INCLUDE_DIRECTORIES( ${DGL_SOURCE_DIR}/include ${DTK_INCLUDE}
		${GLX_HEADER_PATH} ${COIN_INCLUDE_DIR} ${PRODUCER_INCLUDE_DIR}
		${OPENTHREADS_INCLUDE_DIR} )
ADD_EXECUTABLE( dcoin-fly dcoin-fly.cpp )
TARGET_LINK_LIBRARIES( dcoin-fly dcoin )
SET_TARGET_PROPERTIES( dcoin-fly PROPERTIES LINK_FLAGS ${DGL_LIB_FLAGS} )
INSTALL( TARGETS dcoin-fly RUNTIME DESTINATION bin PERMISSIONS OWNER_READ
		OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE WORLD_READ
		WORLD_EXECUTE )
