# CMakeLists.txt to build flockofbirds

INCLUDE_DIRECTORIES( ${DTK_SOURCE_DIR}/include )
ADD_LIBRARY( flock MODULE flock.cpp )
TARGET_LINK_LIBRARIES( flock dtk )
SET_TARGET_PROPERTIES( flock PROPERTIES PREFIX "" )
INSTALL( TARGETS flock LIBRARY DESTINATION etc/dtk/serviceDSO PERMISSIONS
		OWNER_READ OWNER_WRITE OWNER_EXECUTE GROUP_READ GROUP_EXECUTE
		WORLD_READ WORLD_EXECUTE )
INSTALL( FILES flock.config slush.config DESTINATION etc/dtk/serviceDSO
		PERMISSIONS OWNER_READ OWNER_WRITE GROUP_READ WORLD_READ )
